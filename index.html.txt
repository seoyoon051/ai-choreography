<script>
    function generateChoreo() {
        const genre = document.getElementById('genre').value;
        const mood = document.getElementById('mood').value;
        const style = document.getElementById('style').value;
        const resultDiv = document.getElementById('result');

        if(!genre || !mood || !style) {
            alert("모든 항목을 입력해주세요!");
            return;
        }

        const concepts = [
            `당신을 위한 '${genre}' 안무: 전체적으로 ${mood} 흐름을 유지하며, ${style}에 맞춰 절도 있는 포인트 동작을 3번 반복해보세요.`,
            `이번 '${genre}' 스타일은 ${mood}하게 시작해서 ${style} 구간에서 동작을 크게 확장하는 것이 핵심입니다.`,
            `추천 안무: ${genre}의 기본 스텝을 ${mood}한 느낌으로 소화하고, ${style} 타이밍에 맞춰 시선을 고정해보세요.`
        ];

        const randomConcept = concepts[Math.floor(Math.random() * concepts.length)];
        
        resultDiv.style.display = 'block';
        resultDiv.innerText = randomConcept;
    }
</script>
